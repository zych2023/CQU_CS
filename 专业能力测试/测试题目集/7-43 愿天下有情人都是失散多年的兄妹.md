# 题目
呵呵。大家都知道五服以内不得通婚，即两个人最近的共同祖先如果在五代以内（即本人、父母、祖父母、曾祖父母、高祖父母）则不可通婚。本题就请你帮助一对有情人判断一下，他们究竟是否可以成婚？

### 输入格式：

输入第一行给出一个正整数`N`（2 ≤ `N` ≤104），随后`N`行，每行按以下格式给出一个人的信息：

```
本人ID 性别 父亲ID 母亲ID
```

其中`ID`是5位数字，每人不同；性别`M`代表男性、`F`代表女性。如果某人的父亲或母亲已经不可考，则相应的`ID`位置上标记为`-1`。

接下来给出一个正整数`K`，随后`K`行，每行给出一对有情人的`ID`，其间以空格分隔。

注意：题目保证两个人是同辈，每人只有一个性别，并且血缘关系网中没有乱伦或隔辈成婚的情况。

### 输出格式：

对每一对有情人，判断他们的关系是否可以通婚：如果两人是同性，输出`Never Mind`；如果是异性并且关系出了五服，输出`Yes`；如果异性关系未出五服，输出`No`。

# 解答
```cpp
#include <iostream>
#include <set>
#include <queue>
#include <algorithm>
using namespace std;

struct node {
    int f, m, sex;  // f:父亲ID, m:母亲ID, sex:性别(0男1女)
};

node v[100010];       // 存储每个人的信息
int level[100010];    // 记录BFS中的辈分(层数)
bool exist[100010];   // 标记该ID是否存在

int main() {
    int n, m, id, father, mother, a, b;
    cin >> n;  // 读入人数
    char c;    // 性别字符('M'或'F')
    
    for (int i = 0; i < n; i++) {
        cin >> id >> c >> father >> mother;
        v[id].f = father;    // 记录父亲ID
        v[id].m = mother;    // 记录母亲ID
        // 记录性别：'M'为男(0)，'F'为女(1)
        v[id].sex = (c == 'M') ? 0 : 1;
        exist[id] = true;    // 标记该ID存在
        
        // 父母的性别：父亲为男(0)，母亲为女(1)
        if (father != -1) v[father].sex = 0;
        if (mother != -1) v[mother].sex = 1;
    }
    
    cin >> m;  // 读入查询次数
    for (int i = 0; i < m; i++) {
        // 每次查询前重置辈分数组
        fill(level, level + 100010, 0);
        cin >> a >> b;
        
        // 若两人性别相同，直接输出"Never Mind"
        if (v[a].sex == v[b].sex) {
            cout << "Never Mind" << endl;
            continue;
        }
        
        queue<int> q;    // BFS队列
        set<int> s;      // 用于检测共同祖先
        q.push(a);       // 加入第一个人
        q.push(b);       // 加入第二个人
        level[a] = 1;    // 第一个人辈分设为1
        level[b] = 1;    // 第二个人辈分设为1
        int flag = 0;    // 标记是否找到共同祖先
        
        while (!q.empty()) {
            int top = q.front();
            q.pop();
            
            // 尝试插入当前ID到集合，若已存在则说明是共同祖先
            int size_before = s.size();
            s.insert(top);
            if (size_before == s.size()) {  // 插入失败，找到共同祖先
                cout << "No" << endl;
                flag = 1;
                break;
            }
            
            // 若当前ID不存在，跳过其父母
            if (!exist[top]) continue;
            
            // 只追溯到4代以内(辈分<=4)
            if (level[top] <= 4) {
                int fa = v[top].f;  // 父亲ID
                int mo = v[top].m;  // 母亲ID
                
                // 父亲存在则加入队列，辈分+1
                if (fa != -1) {
                    q.push(fa);
                    level[fa] = level[top] + 1;
                }
                // 母亲存在则加入队列，辈分+1
                if (mo != -1) {
                    q.push(mo);
                    level[mo] = level[top] + 1;
                }
            }
        }
        
        // 若未找到共同祖先，输出"Yes"
        if (flag == 0) {
            cout << "Yes" << endl;
        }
    }
    
    return 0;
}
```
