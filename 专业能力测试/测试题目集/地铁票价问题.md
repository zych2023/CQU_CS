# 题目
地铁根据车票种类、乘车距离和时长计算票价。其中，乘坐距离在20公里以内的（含20公里），普通票是4元，学生票是2元；乘坐距离在20公里～50公里之间的（含50公里），普通票是8元，学生票是4元；乘坐距离在50公里以上的，普通票是10元，学生票是5元。如果乘车时长超过180分钟（不含180分钟），则按照该车票种类的最高票价计费。

现对一组地铁乘坐记录，计算其票价。一条地铁乘坐记录包含3个数据项：第1项是车票类型（整数1表示普通票，整数2表示学生票）；第2项是乘车距离（整数表示）；第3项是乘车时长（整数表示，单位：分钟）。用整数表示地铁票价。

### 裁判测试程序样例：

```c++
#include<iostream>
using namespace std;

class SubwayRecord{
public:
    SubwayRecord(int distance, int duration){
        this->distance = distance;
        this->duration = duration;
    }
    virtual ~SubwayRecord(){}
    virtual int fare()=0; // 计算地铁票价
protected:
    int distance;
    int duration;
};

/* 请在这里填写答案 */

int main(){
    int n; //n条地铁乘坐记录
    cin>>n;
    int t; //车票类型（1：普通票；2：学生票）
    int distance; //乘车距离
    int duration; //乘车时长（单位：分钟）
    SubwayRecord* record;
    for(int i=0;i<n;i++){
        cin>>t>>distance>>duration;
        if(t==1){
            record = new RegularSubwayRecord(distance, duration);
        }else{
            record = new StudentSubwayRecord(distance, duration);
        }
        cout<<record->fare()<<endl;
        delete record;
    }
    return 0;
}
```

# 解答
```cpp
// 普通票地铁记录类
class RegularSubwayRecord : public SubwayRecord {
public:
    // 构造函数，调用基类构造函数初始化
    RegularSubwayRecord(int distance, int duration) : SubwayRecord(distance, duration) {}
    
    // 计算普通票票价
    int fare() override {
        // 时长超过180分钟，按最高票价10元计费
        if (duration > 180) {
            return 10;
        }
        // 根据距离计算票价
        if (distance <= 20) {
            return 4;
        } else if (distance <= 50) {
            return 8;
        } else {
            return 10;
        }
    }
};

// 学生票地铁记录类
class StudentSubwayRecord : public SubwayRecord {
public:
    // 构造函数，调用基类构造函数初始化
    StudentSubwayRecord(int distance, int duration) : SubwayRecord(distance, duration) {}
    
    // 计算学生票票价
    int fare() override {
        // 时长超过180分钟，按最高票价5元计费
        if (duration > 180) {
            return 5;
        }
        // 根据距离计算票价
        if (distance <= 20) {
            return 2;
        } else if (distance <= 50) {
            return 4;
        } else {
            return 5;
        }
    }
};
```
