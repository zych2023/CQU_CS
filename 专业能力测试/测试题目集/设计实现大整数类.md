# 题目
设计实现“HugeInteger” 大整数类，实现大整数的加法、输入和输出功能。能够满足以下程序的运行：

```c++
#include <iostream>
#include <string>
using namespace std;

/* 请在这里写出HugeInteger类的定义和实现 */


int main()
{
   HugeInteger n1( 7654321 );
   HugeInteger n2( 7891234 );
   HugeInteger n3( "99999999999999999999999999999" );
   HugeInteger n4( "1" );
   HugeInteger n5;
   cout << "n1 is " << n1 << "\nn2 is " << n2
      << "\nn3 is " << n3 << "\nn4 is " << n4
      << "\nn5 is " << n5 << "\n\n";
   n5 = n1 + n2;
   cout << n1 << " + " << n2 << " = " << n5 << "\n\n";
   cout << n3 << " + " << n4 << "\n= " << ( n3 + n4 ) << "\n\n";
   n5 = n1 + 9;
   cout << n1 << " + " << 9 << " = " << n5 << "\n\n";
   n5 = n2 + "10000";
   cout << n2 << " + " << "10000" << " = " << n5 << endl;
    
    return 0;
}
```

# 解答
```cpp
class HugeInteger {
private:
    string digits; // 存储大整数的字符串，高位在前

public:
    // 默认构造函数，初始化为0
    HugeInteger() : digits("0") {}

    // 整数构造函数，将整数转换为字符串
    HugeInteger(long long num) {
        if (num == 0) {
            digits = "0";
            return;
        }
        bool isNegative = false;
        if (num < 0) {
            isNegative = true;
            num = -num;
        }
        while (num > 0) {
            digits.push_back((num % 10) + '0');
            num /= 10;
        }
        if (isNegative) {
            digits.push_back('-');
        }
        reverse(digits.begin(), digits.end());
    }

    // 字符串构造函数，确保字符串仅包含数字
    HugeInteger(const string& s) {
        // 检查字符串是否合法（仅包含数字）
        for (char c : s) {
            if (c < '0' || c > '9') {
                digits = "0"; // 非法字符串初始化为0
                return;
            }
        }
        digits = s;
        // 去除前导零
        size_t start = digits.find_first_not_of('0');
        if (start != string::npos) {
            digits = digits.substr(start);
        } else {
            digits = "0"; // 全零字符串处理为"0"
        }
    }

    // 加法运算：两个HugeInteger相加
    HugeInteger operator+(const HugeInteger& other) const {
        string result;
        int i = digits.size() - 1;
        int j = other.digits.size() - 1;
        int carry = 0;

        // 逐位相加，处理进位
        while (i >= 0 || j >= 0 || carry > 0) {
            int sum = carry;
            if (i >= 0) {
                sum += digits[i--] - '0';
            }
            if (j >= 0) {
                sum += other.digits[j--] - '0';
            }
            carry = sum / 10;
            result.push_back((sum % 10) + '0');
        }

        // 结果反转，得到正确的顺序
        reverse(result.begin(), result.end());
        return HugeInteger(result);
    }

    // 加法运算：HugeInteger与整数相加
    HugeInteger operator+(long long num) const {
        return *this + HugeInteger(num);
    }

    // 加法运算：HugeInteger与字符串相加
    HugeInteger operator+(const string& s) const {
        return *this + HugeInteger(s);
    }

    // 输出运算符重载
    friend ostream& operator<<(ostream& os, const HugeInteger& hi) {
        os << hi.digits;
        return os;
    }

    // 输入运算符重载
    friend istream& operator>>(istream& is, HugeInteger& hi) {
        string s;
        is >> s;
        hi = HugeInteger(s);
        return is;
    }
};
```
