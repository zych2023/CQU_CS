# 题目
每天都要处理很多事务，为了更好地安排工作，希望在每天开始工作前，根据工作记录，生成工作备忘录。首先输入工作记录数（大于0的一个整数），再逐条输入各条工作记录，每条工作记录包括：工作名，开始时间，结束时间。假设每项工作的开始时间均小于它的结束时间，并且各项工作的开始时间互不相同。

我们的工作是需要把这些工作记录按开始时间排序并输出，在输出时，如果某项工作与若干项工作冲突（在做该项工作时，需要同时做其它工作），则在该工作名前加'*'。

### 函数接口定义：

```c++
Node* add(Node *, Node *);
void display(Node *);
```

### 裁判测试程序样例：

```c++
#include<iostream>
#include <string>
using namespace std;
struct Node{
    string name;
    int start;
    int end;
    Node *next;
};
Node* add(Node *, Node *);
void display(Node *);
bool check(Node *head)
{
    if(head==NULL || head->next==NULL) return true;
    Node *p=head->next;
    if(head->start > p->start) return false;
    return check(p);
}
int main()
{
    Node *head=NULL, *p;
    int i, repeat;
    cin>>repeat;
    for(i=0;i<repeat;i++){
        p = new Node;
        cin>>p->name>>p->start>>p->end;
        p->next=NULL;
        head = add(head, p);
    }
    if(!check(head)) cout<<"ERROR"<<endl;
    display(head);
    return 0;
}
```

# 解答
```cpp
Node* add(Node *head, Node *p) {
    // 若链表为空，直接返回新节点作为头节点
    if (head == NULL) {
        return p;
    }
    // 若新节点的开始时间小于头节点，插入到头部
    if (p->start < head->start) {
        p->next = head;
        return p;
    }
    // 寻找插入位置
    Node *prev = head;
    Node *current = head->next;
    while (current != NULL) {
        if (p->start < current->start) {
            // 插入到prev和current之间
            prev->next = p;
            p->next = current;
            return head;
        }
        prev = current;
        current = current->next;
    }
    // 所有节点均小于新节点，插入到末尾
    prev->next = p;
    return head;
}

void display(Node *head) {
    Node *current = head;
    // 遍历每个工作节点
    while (current != NULL) {
        bool hasConflict = false;
        Node *temp = head;
        // 检查当前工作与其他所有工作是否冲突
        while (temp != NULL) {
            if (temp != current) { // 排除自身
                // 冲突条件：temp的开始时间 < current的结束时间 且 temp的结束时间 > current的开始时间
                if (temp->start < current->end && temp->end > current->start) {
                    hasConflict = true;
                    break; // 找到冲突即可退出
                }
            }
            temp = temp->next;
        }
        // 根据是否冲突输出相应内容
        if (hasConflict) {
            cout << "*" << current->name << " " << current->start << " " << current->end << endl;
        } else {
            cout << current->name << " " << current->start << " " << current->end << endl;
        }
        current = current->next;
    }
}
```
