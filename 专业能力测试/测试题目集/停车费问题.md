# 题目
停车场根据车辆类型、停车时段和时长计算停车费。其中，工作日小型汽车的停车费是每小时3元，一天（24小时）不超过20元；工作日大型汽车的停车费是每小时5元，一天（24小时）不超过35元。节假日小型汽车停车免费，节假日大型汽车的停车费是每小时1元，一天（24小时）不超过5元。

现对一组停车记录，计算其停车费用。一条停车记录包含3个数据项：第1项是汽车类型（整数1表示小型汽车，整数2表示大型汽车）；第2项是停车时段（整数1表示在工作日，整数0表示在节假日）；第3项是停车时长（整数表示，单位：小时）。用整数表示停车费。

### 裁判测试程序样例：

```c++
#include<iostream>
#include<iomanip>
using namespace std;

class ParkingRecord{
public:
    ParkingRecord(int inWeekday, int duration){
        this->inWeekday = inWeekday;
        this->duration = duration;
    }
    virtual ~ParkingRecord(){}
    virtual int fee()=0; // 计算停车费
protected:
    int inWeekday;
    int duration;
};

/* 请在这里填写答案 */

int main(){
    int n; //n条停车记录
    cin>>n;
    int t; //汽车类型（1：小型车；2：大型车）
    int inWeekday; //是否在工作日（1：在工作日；0：在节假日）
    int duration; //停车时长（单位：小时）
    ParkingRecord* record;
    for(int i=0;i<n;i++){
        cin>>t>>inWeekday>>duration;
        if(t==1){
            record = new CarParkingRecord(inWeekday, duration);
        }else{
            record = new TruckParkingRecord(inWeekday, duration);
        }
        cout<<record->fee()<<endl;
        delete record;
    }
    return 0;
}
```

# 解答
```cpp
class CarParkingRecord:public ParkingRecord{
    public:
    CarParkingRecord(int inWeekday, int duration):ParkingRecord(inWeekday,duration){}
    int fee(){
        int ans=0;
        if(!inWeekday)return 0;
        int days=duration/24;
        int hours=duration%24;
        ans=hours*3;
        return 20*days+(ans>=20?20:ans);
    }
};

class TruckParkingRecord:public ParkingRecord{
    public:
    TruckParkingRecord(int inWeekday, int duration):ParkingRecord(inWeekday,duration){}
    int fee(){
        int ans=0;
        int days=duration/24;
        int hours=duration%24;
        if(!inWeekday){
            ans=hours;
            return 5*days+(ans>=5?5:ans);
        }
        else{
            ans=5*hours;
            return 35*days+(ans>=35?35:ans);
        }
    }
};

```
