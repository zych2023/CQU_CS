# 解答
```cpp
#include <iostream>
#include <string>
using namespace std;

class Stud {
private:
    int id;         // 学号
    string name;    // 姓名
    int score;      // 成绩
    static int sum; // 总分（静态成员）
    static int num; // 学生人数（静态成员）
public:
    // 构造函数：初始化学生信息，同时更新总分和人数
    Stud(int i, string n, int s) : id(i), name(n), score(s) {
        sum += s;
        num++;
    }
    
    // 显示学生信息（使用英文逗号）
    void disp() {
        cout << id << "," << name << "," << score << endl;
    }
    
    // 静态成员函数：计算平均分
    static double avg() {
        return static_cast<double>(sum) / num;
    }
};
```
