# 题目
为了缓解交通拥堵问题， 城市在每天早晚高峰的限行时段，推行车辆限行政策。对新能源车（绿牌照）在限行时段没有限行要求；而对于燃油车（蓝牌照），则在限行时段根据车辆号牌最后一位阿拉伯数字进行限行，具体情况是：星期一：1和6 限行； 星期二：2和7 限行； 星期三：3和8限行； 星期四：4和9限行； 星期五：5和0限行；星期六和星期日不限行。  
现对一组车辆通行记录，判断其是否限行。车辆通行记录包含4个数据项：第1项是车辆类型（整数1表示新能源车，整数2表示燃油车）；第2项是车辆号牌最后一位阿拉伯数字；第3项是通行日期（整数1～7分别表示星期一到星期日）；第4项是通行时间（整数1表示在限行时段，整数0表示不在限行时段）。如果限行，输出“`Yes`”；否则，输出“`No`”。

### 裁判测试程序样例：

```c++
#include<iostream>
#include<iomanip>
using namespace std;

class TravelRecord{
public:
    TravelRecord(int plateLastDigit, int dayOfWeek, int inRestrictedPeriod){
        this->plateLastDigit = plateLastDigit;
        this->dayOfWeek = dayOfWeek;
        this->inRestrictedPeriod = inRestrictedPeriod;
    }
    virtual ~TravelRecord(){}
    virtual bool isRestricted()=0; // 判断是否限行
protected:
    int plateLastDigit;
    int dayOfWeek;
    int inRestrictedPeriod;
};

/* 请在这里填写答案 */

int main(){
    int n; //n条通行记录
    cin>>n;
    int t; //汽车类型（1：新能源车；2：燃油车）
    int plateLastDigit; //车辆号牌最后一位阿拉伯数字
    int dayOfWeek; //通行日期（星期几）
    int inRestrictedPeriod; //是否在限行时段（0：不在限行时段；1：在限行时段）
    TravelRecord* record;
    for(int i=0;i<n;i++){
        cin>>t>>plateLastDigit>>dayOfWeek>>inRestrictedPeriod;

        }

            cout<<"Yes"<<endl;
        } else {
            cout<<"No"<<endl;
        }
        delete record;
    }
 
    return 0;
}
```

# 解答
```cpp
// 新能源车通行记录类
class NewEnergyVehicleTravelRecord : public TravelRecord {
public:
    // 构造函数，调用基类构造函数初始化
    NewEnergyVehicleTravelRecord(int plateLastDigit, int dayOfWeek, int inRestrictedPeriod)
        : TravelRecord(plateLastDigit, dayOfWeek, inRestrictedPeriod) {}
    
    // 新能源车不限行，始终返回false
    bool isRestricted() override {
        return false;
    }
};

// 燃油车通行记录类
class FuelVehicleTravelRecord : public TravelRecord {
public:
    // 构造函数，调用基类构造函数初始化
    FuelVehicleTravelRecord(int plateLastDigit, int dayOfWeek, int inRestrictedPeriod)
        : TravelRecord(plateLastDigit, dayOfWeek, inRestrictedPeriod) {}
    
    // 根据规则判断是否限行
    bool isRestricted() override {
        // 不在限行时段，不限行
        if (inRestrictedPeriod == 0) {
            return false;
        }
        // 周六或周日，不限行
        if (dayOfWeek == 6 || dayOfWeek == 7) {
            return false;
        }
        // 根据星期几和车牌最后一位判断是否限行
        switch (dayOfWeek) {
            case 1: // 周一：1或6限行
                return (plateLastDigit == 1 || plateLastDigit == 6);
            case 2: // 周二：2或7限行
                return (plateLastDigit == 2 || plateLastDigit == 7);
            case 3: // 周三：3或8限行
                return (plateLastDigit == 3 || plateLastDigit == 8);
            case 4: // 周四：4或9限行
                return (plateLastDigit == 4 || plateLastDigit == 9);
            case 5: // 周五：5或0限行
                return (plateLastDigit == 5 || plateLastDigit == 0);
            default: // 其他情况（理论上不会发生）
                return false;
        }
    }
};
```
